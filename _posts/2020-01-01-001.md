#include <iostream>

int main(void)
{
    std::cin.tie(0);
    std::ios_base::sync_with_stdio(false);

    int T;
    std::cin >> T;

    while (T--)
    {
        int x, y;
        std::cin >> x >> y;

        if( x == y ){
            std::cout << x + y << "\n";
        } else {
            std::cout << 2*std::max(x,y) - 1 << "\n";
        }
    }

    return 0;
}
